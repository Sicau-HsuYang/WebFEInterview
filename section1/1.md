## 2、编写一个方法，去掉数组里面重复的内容 var arr = ['abc', 'abcd', 'sss', '2', 'd', 't', '2', 'ss', 'f','22', 'd' ]

```js
function deRepeat(input) {
  let de_repeat = [];
  input.forEach(item => {
    if(!de_repeat.includes(item)) {
      de_repeat.push(item);
    }
  })
  return de_repeat;
}
```
